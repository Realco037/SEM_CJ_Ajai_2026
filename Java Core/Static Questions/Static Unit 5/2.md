import java.util.Scanner;

class VowelConsonantPrinter {
    private final Object lock = new Object();
    private final int limit;
    private boolean vowelTurn = true;
    private int index = 0;

    public VowelConsonantPrinter(int limit) {
        this.limit = limit;
    }

    private char getVowel(int i) {
        char[] vowels = {'A', 'E', 'I', 'O', 'U'};
        return vowels[i % vowels.length];
    }

    private char getConsonant(int i) {
        char[] consonants = {
            'B', 'C', 'D', 'F', 'G', 'H', 'J', 'K', 'L', 'M',
            'N', 'P', 'Q', 'R', 'S', 'T', 'V', 'W', 'X', 'Y', 'Z'
        };
        return consonants[i % consonants.length];
    }

    public void printVowel() {
        for (int i = 0; i < limit; i++) {
            synchronized (lock) {
                while (!vowelTurn) {
                    try { lock.wait(); } catch (InterruptedException e) {}
                }
                System.out.print(getVowel(i));
                System.out.print(" ");
                vowelTurn = false;
                lock.notify();
            }
        }
    }

    public void printConsonant() {
        for (int i = 0; i < limit; i++) {
            synchronized (lock) {
                while (vowelTurn) {
                    try { lock.wait(); } catch (InterruptedException e) {}
                }
                System.out.print(getConsonant(i));
                if (i != limit - 1) System.out.print(" ");
                vowelTurn = true;
                lock.notify();
            }
        }
    }
}
